\newif\ifUrlListe@

\newcommand\saved@hyper@linkurl{}
\let\saved@hyper@linkurl\hyper@linkurl
\renewcommand{\hyper@linkurl}[2]{%
  \ifUrlListe@
  \else
    \begingroup
      \refstepcounter{UrlListe}%
      \protected@edef\UrlListe@Text{#1}%
      \@onelevel@sanitize\UrlListe@Text
      \EdefEscapeHex\UrlListe@HexLink{#2}%
      \zref@labelbylist{\theUrlListe}{UrlListe}%
    \endgroup
  \fi
  \saved@hyper@linkurl{#1}{#2}%
}

\newcommand*{\UrlListe}{%
\clearpage
\newgeometry{left=2cm,right=2cm,top=2cm,bottom=3cm}
\fancyput(-2.54cm,-25.96cm){\includegraphics[width=21cm,height=28.5cm]{logos_images/imagedefondrose.jpg}}%
  \zref@refused{LastPage}%
  \edef\UrlListe@Max{%
    \zref@extractdefault{LastPage}{UrlListe@Max}{-1}%
  }%
  \ifnum\UrlListe@Max<0 %
    \@latex@warning@no@line{Il faut compiler deux fois pour avoir la liste des liens ;)}%
  \else
    \UrlListe@true
    \begin{UrlListeEnv}%
      \let\UrlListe@LastPage\@empty
      \setcounter{UrlListeAux}{0}%
      \@whilenum\value{UrlListeAux}<\UrlListe@Max\do{%
        \stepcounter{UrlListeAux}%
        \zref@refused{\theUrlListeAux}%
        \zref@ifrefundefined{\theUrlListeAux}{%
        }{%
          \EdefUnescapeHex\UrlListe@Link{%
            \zref@extract{\theUrlListeAux}{UrlListe@HexLink}%
          }%
          \zref@def@extract\UrlListe@Text{\theUrlListeAux}{UrlListe@Text}%
          \zref@def@extract\UrlListe@Page{\theUrlListeAux}{page}%
          \edef\UrlListe@Next{%
            \noexpand\UrlListeItem{%
              \etex@unexpanded\expandafter{\UrlListe@Page}%
            }{%
              \etex@unexpanded\expandafter{\UrlListe@Link}%
            }{%
              \etex@unexpanded\expandafter{\UrlListe@Text}%
            }%
          }%
          \ifx\UrlListe@Page\UrlListe@LastPage
          \else
            \expandafter\UrlListePage\expandafter{\UrlListe@Page}%
            \let\UrlListe@LastPage\UrlListe@Page
          \fi
          \UrlListe@Next
        }%
      }%
    \end{UrlListeEnv}%
  \fi
}
